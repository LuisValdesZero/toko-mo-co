from openai import OpenAI

client = OpenAI(
    base_url="http://localhost:8080/v1",
    api_key="your-openai-key",
    default_headers={
        "X-Agent-ID":   "my-agent",    # Agents view
        "X-App-Name":   "my-project",  # groups agents by app
        "X-Session-ID": "session-123", # groups into sessions
    }
)

response = client.chat.completions.create(
    model="gpt-4o",
    messages=[{"role": "user", "content": "Hello!"}]
)
Agent tagging — add X-Agent-ID and X-App-Name to see named agents in the Agents view with per-agent cost, token, and request breakdowns. Future Rules Engine will use these headers to apply per-agent policies. 
import anthropic

client = anthropic.Anthropic(
    base_url="http://localhost:8080",
    api_key="your-anthropic-key",
    default_headers={
        "X-Agent-ID": "my-agent",
        "X-App-Name": "my-project",
    }
)

message = client.messages.create(
    model="claude-opus-4-5-20250929",
    max_tokens=1024,
    messages=[{"role": "user", "content": "Hello!"}]
)
